40def inferDateFormat(date_string:str) -> date:
41 formats=['%m/%d/%y', '%m/%d/%Y', '%m-%d-%y', '%m-%d-%Y', '%Y-%m-%d']
42 d:date = None
43 for f in formats:
44 try:
45 d=datetime.strptime(date_string,f)
46 return d
47 except:
48 pass # try try again
49 # give up with a nicely formatted date, even if wrong
50 log.error("Unable to parse date out of '{date}'. Returning now.".format(date=date))
51 return datetime.now()

213class Uncertain():
214 '''
215 Uncertain is a mixin class that adds a confidence value (0.0-1.0) to a Node or Edge.
216 This is implemented to store the float value and generate description qualitative
217 text clauses for the ranges of this confidence value.
218 '''
219 confidence = nm.FloatProperty(default=1.0)
220 def uncertainty_to_text(self) -> str:
221 if self.confidence < 0.05:
222 return "with almost no chance"
223 if self.confidence >= 0.05 and self.confidence < 0.20:
224 return "very unlikely"
225 if self.confidence >= 0.2 and self.confidence < 0.45:
226 return "unlikely"
227 if self.confidence >= 0.45 and self.confidence < 0.55:
228 return "with roughly even chance"
229 if self.confidence >= 0.55 and self.confidence < 0.8:
230 return "likely"
231 if self.confidence >= 0.8 and self.confidence < 0.95:
232 return "very likely"
233 if self.confidence >= 0.55 and self.confidence < 1.0:
234 return "with almost certainty"
235 return "with certainty"

237class Scored():
238 '''
239 Scored is a mixin class that adds a score value (0.0-1.0) to a Node or Edge.
240 This is implemented to store the float value and generate descriptive qualitative
241 text clauses for the ranges of this score value.
242 '''
243 score = nm.FloatProperty(default=-1.0)
244 def score_to_text(self) -> str:
245 if self.score < 0.0:
246 return "unknown"
247 if self.score < 0.15:
248 return "very low"
249 if self.score >= 0.15 and self.score < 0.40:
250 return "low"
251 if self.score >= 0.4 and self.score < 0.6:
252 return "medium"
253 if self.score >= 0.6 and self.score < 0.85:
254 return "high"
255 if self.score >= 0.85 and self.score < 1.0:
256 return "very high"
257 return "perfect"

656 date=self.date)

686 how relevant that entry is with respect to the request. These edges are generally created

688
689 Feedback clicked ("clicked" property is a date) indicates when the last time the source
690 link for the TLDR entry was clicked by the user.
691 '''

693
694 about_entry = nm.RelationshipTo(TldrEntry, 'ABOUT', model=FeedbackForEntry, cardinality=One)
695 from_request = nm.RelationshipTo(Request, 'FROM', model=FeedbackForRequest, cardinality=One)

697 def to_text(self) -> str:
698 click_status = "not clicked"

700 click_status= "clicked on {date}".format(date=self.click_date.strftime("%m/%d/%Y"))

704 rate_status = "rated as {rate}".format(rate=self.score_to_text())
705

730 '''
731 EvalKey is an indication of the relationship between a Content (e.g., a news article) and a
732 Request (i.e., what the user is interested in). This represents a "correct" tailoring pair.
733 This also includes a score and text to help identify what was most relevant in the pair.
734 '''

736 key_for_content = nm.RelationshipTo(Content, 'KEY_FOR_CONTENT', model=KeyForContent, cardinality=One)
737 key_for_request = nm.RelationshipTo(Request, 'KEY_FOR_REQUEST', model=KeyForRequest, cardinality=One)
