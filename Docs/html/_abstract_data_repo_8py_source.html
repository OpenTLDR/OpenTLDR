39 def parseAnnotatedText(cls,text:str ) -> tuple[dict, str]:
40 '''
41 Parses a string from an annotated text file, extracts the annotations, and
42 returns a tuple (dict of metadata , string of non annotation text block).
43
44 Annotations are commented property key : value pairs, such as:
45 # uid = 1234\n
46 This is a text block.\n
47 # name = Chris\n
48 Another line.\n
49 And one at the end.\n
50
51 Will return ({'uid': '1234', 'name':'Chris}, 'This is a text block.\nAnother line.\nAnd one at the end')
52 '''
53 # Identifies a property key
54 property_regex=re.compile(r'^#\s*[a-z_]+\s*\:\s*')
55
56 content_text=""
57 properties={}
58
59 for line in text.splitlines():
60 match= property_regex.search(line)
61 if match:
62 # This line is an annotation
63 key = re.sub(r'[#:\s]',"",match.group()).lower()
64 value = ''.join(re.split(property_regex,line)).strip()
65 properties[key]=value
66 else:
67 # This line is a text block
68 content_text='\n'.join([content_text,line])
69
70 return (properties, content_text.strip())
